// ‚úÖ INICIALIZACI√ìN AL CARGAR LA P√ÅGINA
document.addEventListener('DOMContentLoaded', function() {
    initializeWelcomePage();
});

function initializeWelcomePage() {
    // ‚úÖ Cargar tema guardado REQUERIDO
    Themes.loadSavedTheme();
    
    // ‚úÖ Configurar formulario nombre OBLIGATORIO
    setupCustomerForm();
    
    // ‚úÖ Configurar cambio de tema
    setupThemeToggle();
    
    // Verificar si ya hay un nombre guardado
    checkExistingCustomer();
}

// ‚úÖ CONFIGURAR FORMULARIO NOMBRE CLIENTE
function setupCustomerForm() {
    const form = document.getElementById('customer-form');
    const nameInput = document.getElementById('customer-name');
    const continueBtn = document.getElementById('continue-btn');
    const errorDiv = document.getElementById('name-error');
    
    // Validaci√≥n en tiempo real
    nameInput.addEventListener('input', function() {
        const name = this.value.trim();
        const isValid = validateCustomerName(name);
        
        // ‚úÖ BOT√ìN SOLO ACTIVO CON NOMBRE V√ÅLIDO
        continueBtn.disabled = !isValid;
        
        if (name.length > 0 && !isValid) {
            showNameError('Por favor, ingresa solo letras y espacios');
        } else {
            hideNameError();
        }
    });
    
    // ‚úÖ SUBMIT FORMULARIO
    form.addEventListener('submit', function(e) {
        e.preventDefault();
        
        const name = nameInput.value.trim();
        
        if (validateCustomerName(name)) {
            // ‚úÖ GUARDAR NOMBRE Y CONTINUAR
            saveCustomerName(name);
            redirectToProducts();
        } else {
            showNameError('Por favor, ingresa un nombre v√°lido');
        }
    });
}

// ‚úÖ VALIDAR NOMBRE CLIENTE
function validateCustomerName(name) {
    // Solo letras, espacios y acentos
    const nameRegex = /^[a-zA-Z√°√©√≠√≥√∫√Å√â√ç√ì√ö√±√ë\s]{2,50}$/;
    return name.length >= 2 && nameRegex.test(name);
}

// ‚úÖ GUARDAR NOMBRE EN LOCALSTORAGE
function saveCustomerName(name) {
    localStorage.setItem(CONFIG.LOCAL_STORAGE_KEYS.CUSTOMER_NAME, name);
    console.log('Nombre del cliente guardado:', name);
}

// ‚úÖ REDIRIGIR A PRODUCTOS
function redirectToProducts() {
    // Animaci√≥n suave antes de redirigir
    document.body.classList.add('fade-out');
    
    setTimeout(() => {
        window.location.href = './products.html';
    }, 300);
}

// Verificar si ya hay un cliente logueado
function checkExistingCustomer() {
    const savedName = localStorage.getItem(CONFIG.LOCAL_STORAGE_KEYS.CUSTOMER_NAME);
    
    if (savedName) {
        const nameInput = document.getElementById('customer-name');
        const continueBtn = document.getElementById('continue-btn');
        
        nameInput.value = savedName;
        continueBtn.disabled = false;
        
        // Mostrar mensaje de bienvenida
        showWelcomeBackMessage(savedName);
    }
}

function showWelcomeBackMessage(name) {
    const welcomeHeader = document.querySelector('.welcome-header h2');
    welcomeHeader.textContent = `¬°Bienvenido de nuevo, ${name}!`;
}

// ‚úÖ CONFIGURAR TOGGLE TEMA
function setupThemeToggle() {
    const themeBtn = document.getElementById('theme-toggle');
    
    themeBtn.addEventListener('click', function() {
        Themes.toggleTheme();
        updateThemeButtonText();
    });
    
    // Actualizar texto inicial
    updateThemeButtonText();
}

function updateThemeButtonText() {
    const themeBtn = document.getElementById('theme-toggle');
    const icon = themeBtn.querySelector('i');
    const text = themeBtn.querySelector('span');
    
    const currentTheme = document.documentElement.getAttribute('data-theme');
    
    if (currentTheme === 'dark') {
        icon.className = 'fas fa-sun';
        text.textContent = 'Tema Claro';
    } else {
        icon.className = 'fas fa-moon';
        text.textContent = 'Tema Oscuro';
    }
}

// Mostrar/ocultar errores de nombre
function showNameError(message) {
    const errorDiv = document.getElementById('name-error');
    errorDiv.textContent = message;
    errorDiv.style.display = 'block';
}

function hideNameError() {
    const errorDiv = document.getElementById('name-error');
    errorDiv.style.display = 'none';
}

function setupProductCardEvents(card, product) {
    const plusBtn = card.querySelector('.qty-btn.plus');
    const minusBtn = card.querySelector('.qty-btn.minus');
    const addToCartBtn = card.querySelector('.add-to-cart-btn');
    const viewDetailsBtn = card.querySelector('.view-details-btn');
    const quantitySpan = card.querySelector('.quantity');
    
    let quantity = 1;
    
    // ‚úÖ Controles cantidad +/-
    plusBtn.addEventListener('click', () => {
        quantity++;
        quantitySpan.textContent = quantity;
    });
    
    minusBtn.addEventListener('click', () => {
        if (quantity > 1) {
            quantity--;
            quantitySpan.textContent = quantity;
        }
    });
    
    // ‚úÖ AGREGAR AL CARRITO
    addToCartBtn.addEventListener('click', () => {
        Cart.addProduct(product, quantity);
        showAddedToCartMessage(product.name, quantity);
        
        // Resetear cantidad
        quantity = 1;
        quantitySpan.textContent = quantity;
    });
    
    // Ver detalles en modal
    viewDetailsBtn.addEventListener('click', () => {
        showProductModal(product);
    });
}

function createPagination(totalPages) {
    const paginationContainer = document.getElementById('pagination');
    
    if (totalPages <= 1) {
        paginationContainer.innerHTML = '';
        return;
    }
    
    let paginationHTML = '<div class="pagination-controls">';
    
    // Bot√≥n anterior
    if (currentPage > 1) {
        paginationHTML += `
            <button class="page-btn" data-page="${currentPage - 1}">
                <i class="fas fa-chevron-left"></i> Anterior
            </button>
        `;
    }
    
    // N√∫meros de p√°gina
    for (let i = 1; i <= totalPages; i++) {
        const isActive = i === currentPage ? 'active' : '';
        paginationHTML += `
            <button class="page-btn ${isActive}" data-page="${i}">
                ${i}
            </button>
        `;
    }
    
    // Bot√≥n siguiente
    if (currentPage < totalPages) {
        paginationHTML += `
            <button class="page-btn" data-page="${currentPage + 1}">
                Siguiente <i class="fas fa-chevron-right"></i>
            </button>
        `;
    }
    
    paginationHTML += '</div>';
    paginationContainer.innerHTML = paginationHTML;
    
    // ‚úÖ CONFIGURAR EVENTOS PAGINACI√ìN
    setupPaginationEvents();
}

function setupPaginationEvents() {
    const pageButtons = document.querySelectorAll('.page-btn');
    
    pageButtons.forEach(btn => {
        btn.addEventListener('click', function() {
            const page = parseInt(this.dataset.page);
            if (page !== currentPage) {
                currentPage = page;
                displayProducts();
                
                // Scroll suave al inicio de productos
                document.querySelector('.products-grid-container').scrollIntoView({
                    behavior: 'smooth'
                });
            }
        });
    });
}
